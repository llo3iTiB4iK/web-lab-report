function checkBrackets(str) {
    const stack = [];
    const openBrackets = ['(', '{', '['];
    const closeBrackets = [')', '}', ']'];
    for (let char of str) {
        if (openBrackets.includes(char)) {
            stack.push(char);
        } else if (closeBrackets.includes(char)) {
            if (stack.length === 0) { return false; }
            if (stack.pop() !== openBrackets[closeBrackets.indexOf(char)]) {
                return false;
            }
        }
    }
    return stack.length === 0;
}

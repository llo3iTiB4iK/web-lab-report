function countTags(posts) {
    const tagCounts = {};
    for (let post of posts) {
        for (let tag of post.tags) {
            if (tag in tagCounts) {
                tagCounts[tag] += 1;
            } else {
                tagCounts[tag] = 1;
            }
        }
    }
    return tagCounts;
}
